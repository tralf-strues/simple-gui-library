52 void ButtonBaseSkinEventListener::onEvent(Sml::Event* event)
53 {
54 ASSERT_CATEGORY(Sml::MouseEvent);
55 ButtonBaseSkin& skin = static_cast<ButtonBaseSkin&>(*button.getSkin());
56
57 switch (event->getType())
58 {
59 case MouseEnteredEvent::getStaticType():
60 {
61 skin.applyInteractionStyle(ButtonBaseSkin::InteractionStyle::Type::HOVERED);
62 break;
63 }
64
65 case MouseExitedEvent::getStaticType():
66 {
67 skin.applyInteractionStyle(ButtonBaseSkin::InteractionStyle::Type::IDLE);
68 break;
69 }
70
71 case Sml::MouseButtonPressedEvent::getStaticType():
72 {
73 if (button.getOnAction() != nullptr)
74 {
75 // FIXME: Add event queue!
76 ActionEvent event{&button};
77 button.getOnAction()->onActionPerformed(&event);
78 }
79
80 break;
81 }
82
83 default:
84 {
85 INVALID_EVENT_TYPE(ButtonBaseSkinEventListener);
86 break;
87 }
88 }
89 }

107 void ButtonBaseSkin::attach()
108 {
109 m_Handler = new ButtonBaseSkinEventListener(*m_Button);
110 m_Button->getEventDispatcher()->attachHandler(BUTTON_SKIN_LISTENER_EVENTS, m_Handler);
111
112 m_Button->addChild(&m_Icon);
113 m_Button->addChild(&m_Text);
114
115 applyStaticStyle();
116 applyInteractionStyle(InteractionStyle::Type::IDLE);
117 }

119 void ButtonBaseSkin::dispose()
120 {
121 m_Button->getEventDispatcher()->detachHandler(m_Handler);
122 delete m_Handler;
123
124 m_Button->removeChild(&m_Icon);
125 m_Button->removeChild(&m_Text);
126 }

152 int32_t ButtonBaseSkin::computePrefHeight(int32_t width) const
153 {
154 const Insets& padding = m_Button->getPadding();
155
156 return padding.top + padding.bottom +
157 std::max(m_Text.computePrefHeight(width), m_Icon.computePrefHeight(width));
158 }

160 int32_t ButtonBaseSkin::computePrefWidth(int32_t height) const
161 {
162 const Insets& padding = m_Button->getPadding();
163
164 int32_t textPrefWidth = m_Text.computePrefWidth(height);
165 int32_t iconPrefWidth = m_Icon.computePrefWidth(height);
166
167 return padding.left + padding.right +
168 ((textPrefWidth != 0 && iconPrefWidth != 0) ? getMargin() : 0) +
169 textPrefWidth + iconPrefWidth;
170 }

172 void ButtonBaseSkin::layoutChildren()
173 {
174 m_Text.setLayoutWidth(m_Text.computePrefWidth());
175 m_Text.setLayoutHeight(m_Text.computePrefHeight());
176
177 m_Icon.setLayoutWidth(m_Icon.computePrefWidth());
178 m_Icon.setLayoutHeight(m_Icon.computePrefHeight());
179
180 const Insets& padding = m_Button->getPadding();
181 Sml::Rectangle<int32_t> contentArea = m_Button->getOriginBounds();
182 contentArea.pos += {padding.left, padding.top};
183 contentArea.width -= padding.left + padding.right;
184 contentArea.height -= padding.top + padding.bottom;
185
186 int32_t centerY = contentArea.pos.y + contentArea.height / 2;
187
188 m_Icon.setLayoutX(contentArea.pos.x);
189 m_Icon.setLayoutY(centerY - m_Icon.getLayoutHeight() / 2);
190
191 m_Text.setLayoutX(contentArea.pos.x + contentArea.width - m_Text.getLayoutWidth());
192 m_Text.setLayoutY(centerY - m_Text.getLayoutHeight() / 2);
193 }

200 void ButtonBaseSkin::applyStaticStyle()
201 {
202 if (m_StaticStyle != nullptr)
203 {
204 m_Button->setPadding(m_StaticStyle->padding);
205 m_Button->setBorder(&m_StaticStyle->border);
206 }
207 else
208 {
209 m_Button->setPadding(Insets::EMPTY);
210 m_Button->setBorder(nullptr);
211 }

214 void ButtonBaseSkin::applyInteractionStyle(InteractionStyle::Type type)
215 {
216 const InteractionStyle* style = nullptr;
217 switch (type)
218 {
219 case InteractionStyle::Type::IDLE: { style = m_IdleStyle; break; }
220 case InteractionStyle::Type::HOVERED: { style = m_HoveredStyle; break; }
221 case InteractionStyle::Type::PRESSED: { style = m_PressedStyle; break; }
222 }
223
224 if (style != nullptr)
225 {
226 m_Text.setColor(style->foreground);
227 m_Button->setBackground(style->background);
228 }
229 else
230 {
231 m_Text.setColor(Sml::COLOR_BLACK);
232 m_Button->setBackground(nullptr);
233 }
234 }
