32 class ButtonBaseSkinEventListener : public ComponentEventListener<Button>
33 {
34 public:
35 DEFINE_STATIC_LISTENED_EVENT_TYPES(MouseEnteredEvent::getStaticType(),
36 MouseExitedEvent::getStaticType(),
37 Sml::MouseButtonPressedEvent::getStaticType())
38
39 public:
40 ButtonBaseSkinEventListener(Button* button) : ComponentEventListener<Button>(button) {}
41
42 virtual void onEvent(Sml::Event* event) override
43 {
44 ASSERT_CATEGORY(Sml::MouseEvent);
45 ButtonBaseSkin& skin = dynamic_cast<ButtonBaseSkin&>(*getComponent()->getSkin());
46
47 switch (event->getType())
48 {
49 case MouseEnteredEvent::getStaticType():
50 {
51 skin.applyInteractionStyle(ButtonBaseSkin::InteractionStyle::Type::HOVERED);
52 break;
53 }
54
55 case MouseExitedEvent::getStaticType():
56 {
57 skin.applyInteractionStyle(ButtonBaseSkin::InteractionStyle::Type::IDLE);
58 break;
59 }
60
61 case Sml::MouseButtonPressedEvent::getStaticType():
62 {
63 if (getComponent()->getOnAction() != nullptr)
64 {
65 // FIXME: Add event queue!
66 ActionEvent event{getComponent()};
67 getComponent()->getOnAction()->onAction(&event);
68 }
69
70 break;
71 }
72
73 default:
74 {
75 INVALID_EVENT_TYPE(ButtonBaseSkinEventListener);
76 break;
77 }
78 }
79
80 event->consume();
81 }
82 };

105 void ButtonBaseSkin::attach(Button* button)
106 {
107 m_Button = button;
108
109 m_Handler = new ButtonBaseSkinEventListener(m_Button);
110 m_Button->getEventDispatcher()->attachHandler(ButtonBaseSkinEventListener::EVENT_TYPES, m_Handler);
111
112 m_Button->addChild(&m_Icon);
113 m_Button->addChild(&m_Text);
114
115 applyStaticStyle();
116 applyInteractionStyle(InteractionStyle::Type::IDLE);
117 }
118
119 void ButtonBaseSkin::dispose()
120 {
121 m_Button->getEventDispatcher()->detachHandler(m_Handler);
122 delete m_Handler;
123
124 m_Button->removeChild(&m_Icon);
125 m_Button->removeChild(&m_Text);
126 }

149 int32_t ButtonBaseSkin::computePrefHeight(int32_t width) const
150 {
151 Insets insets = m_Button->getInsets();
152
153 return insets.top + insets.bottom +
154 std::max(m_Text.computePrefHeight(width), m_Icon.computePrefHeight(width));
155 }

157 int32_t ButtonBaseSkin::computePrefWidth(int32_t height) const
158 {
159 Insets insets = m_Button->getInsets();
160
161 int32_t textPrefWidth = m_Text.computePrefWidth(height);
162 int32_t iconPrefWidth = m_Icon.computePrefWidth(height);
163
164 return insets.left + insets.right +
165 ((textPrefWidth != 0 && iconPrefWidth != 0) ? getMargin() : 0) +
166 textPrefWidth + iconPrefWidth;
167 }

187

189
190 m_Icon.setLayoutX(contentArea.pos.x);
191 m_Icon.setLayoutY(centerY - m_Icon.getLayoutHeight() / 2);
192
193 m_Text.setLayoutX(contentArea.pos.x + contentArea.width - m_Text.getLayoutWidth());
194 m_Text.setLayoutY(centerY - m_Text.getLayoutHeight() / 2);

202 void ButtonBaseSkin::applyStaticStyle()
203 {
204 if (m_StaticStyle != nullptr)
205 {
206 m_Button->setPadding(m_StaticStyle->padding);
207 m_Button->setBorder(&m_StaticStyle->border);
208 }
209 else

212 m_Button->setBorder(nullptr);

216 void ButtonBaseSkin::applyInteractionStyle(InteractionStyle::Type type)
217 {
218 const InteractionStyle* style = nullptr;
219 switch (type)
220 {
221 case InteractionStyle::Type::IDLE: { style = m_IdleStyle; break; }
222 case InteractionStyle::Type::HOVERED: { style = m_HoveredStyle; break; }
223 case InteractionStyle::Type::PRESSED: { style = m_PressedStyle; break; }
224 }
225
226 if (style != nullptr)
227 {
228 m_Text.setColor(style->foreground);
229 m_Button->setBackground(style->background);
230 }
231 else
232 {
233 m_Text.setColor(Sml::COLOR_BLACK);
234 m_Button->setBackground(nullptr);
235 }
236 }
