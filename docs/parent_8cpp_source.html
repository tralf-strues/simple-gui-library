71 int32_t Parent::computePrefWidth(int32_t height)
72 {
73 CHECK_ENABLED_SIZE(m_PrefWidth);
74
75 int32_t minX = INT32_MAX;
76 int32_t maxX = 0;
77 for (Component* child : m_Children)
78 {
79 int32_t childX = child->getLayoutX();
80 int32_t childWidth = child->computePrefWidth();
81
82 if (childWidth == UNLIMITED_SIZE)
83 {
84 return UNLIMITED_SIZE;
85 }
86
87 minX = std::min(minX, childX);
88 maxX = std::max(maxX, childX + childWidth);
89 }
90
91 if (minX != INT32_MAX)
92 {
93 return maxX - minX;
94 }
95
96 return 0;
97 }
