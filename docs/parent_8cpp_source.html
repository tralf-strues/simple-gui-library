15 {

252 void Parent::updateSnapshotSize()
253 {
254 int32_t width = m_LayoutBounds.width;
255 int32_t height = m_LayoutBounds.height;
256
257 if (width <= 0 || height <= 0)
258 {
259 return;
260 }
261
262 if (m_Snapshot == nullptr)
263 {
264 m_Snapshot = new Sml::Texture(width, height);
265 }
266 else if (static_cast<int32_t>(m_Snapshot->getWidth()) < width ||
267 static_cast<int32_t>(m_Snapshot->getHeight()) < height)
268 {
269 delete m_Snapshot;
270 m_Snapshot = new Sml::Texture(width, height);
271 }
272 }
