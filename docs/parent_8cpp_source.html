204 void Parent::updateSnapshotSize()
205 {
206 int32_t width = m_LayoutBounds.width;
207 int32_t height = m_LayoutBounds.height;
208
209 if (width <= 0 || height <= 0)
210 {
211 return;
212 }
213
214 if (m_Snapshot == nullptr)
215 {
216 m_Snapshot = new Sml::Texture(getContextRenderer(), width, height);
217 }
218 else if (static_cast<int32_t>(m_Snapshot->getWidth()) < width ||
219 static_cast<int32_t>(m_Snapshot->getHeight()) < height)
220 {
221 delete m_Snapshot;
222 m_Snapshot = new Sml::Texture(getContextRenderer(), width, height);
223 }
224 }
